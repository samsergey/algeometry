-- | Type family for Clifford algebras with given signature.
type family Cl (p :: Nat) (q :: Nat) (r :: Nat) where
  Cl 0 0 0 = Double
  Cl 0 n 0 = Outer n
  Cl n 0 0 = VGA n
  Cl 2 1 0 = PGA 2
  Cl 3 1 0 = PGA 3
  Cl n 1 0 = PGA n
  Cl p q r = CA p q r
